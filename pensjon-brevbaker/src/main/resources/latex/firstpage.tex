    \logo

}

\newcommand{\firstpage}{
    \thispagestyle{firstpage} %set pagestyle for this page
    \pagestyle{normalpage} % set default pagestyle
    \primaryFont\fontsize{\standardtextsize}{\standardtextspacing}\selectfont
}

\newcommand{\letterdate}{
    \newlength{\letterdatewidth}
    \setlength{\letterdatewidth}{40mm}
    \newlength{\letterdatexpos}
    \setlength{\letterdatexpos}{\dimexpr(\lettermargin+\textwidth-\letterdatewidth)}
    \begin{textblock*}{\letterdatewidth}(\letterdatexpos,80mm)
        \primaryFont\fontsize{\standardtextsize}{\standardtextspacing}\selectfont\hfill~\feltdato
    \end{textblock*}
}
