% removes whitespace between items, removes parskip whitespace over and under the list
% adds custom space to margin and after list and penalty for breaking up an itemlist at the top
\setlist[itemize]{noitemsep, beginpenalty=3000, topsep=0pt, leftmargin=15pt, after=\addvspace{10pt}}
\newenvironment{letteritemize}[1]{
    % remove paragraph spacing if previous element was a paragraph
    \ifdefstring{\previouselement}{paragraph}{\vspace{-\parskip}}{}
    \begin{itemize}
        #1
    } {
    \end{itemize}
    \setpreviouselement{item}
}
